package com.example.drawapp;

import android.graphics.Canvas;
import android.graphics.Color;
import android.graphics.Paint;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    DrawView drawView;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        drawView = findViewById(R.id.drawView);

        findViewById(R.id.btnLine).setOnClickListener(v -> drawView.setShape("line"));
        findViewById(R.id.btnCircle).setOnClickListener(v -> drawView.setShape("circle"));
        findViewById(R.id.btnRect).setOnClickListener(v -> drawView.setShape("rect"));
        findViewById(R.id.btnSquare).setOnClickListener(v -> drawView.setShape("square"));
        findViewById(R.id.btnText).setOnClickListener(v -> drawView.setShape("text"));
    }

    public static class DrawView extends View {

        Paint paint = new Paint();
        String shape = "";

        public DrawView(android.content.Context context, android.util.AttributeSet attrs) {
            super(context, attrs);
            paint.setStrokeWidth(5);
            paint.setStyle(Paint.Style.STROKE);
            paint.setTextSize(80);
        }

        public void setShape(String s){
            shape = s;
            invalidate();
        }

        @Override
        protected void onDraw(Canvas canvas) {
            super.onDraw(canvas);

            paint.setColor(Color.WHITE);
            canvas.drawRect(10,10,getWidth()-10,getHeight()-10,paint);

            if(shape.equals("line")){
                paint.setColor(Color.CYAN);
                canvas.drawLine(100,300,getWidth()-100,getHeight()-300,paint);
            }
            else if(shape.equals("circle")){
                paint.setColor(Color.MAGENTA);
                canvas.drawCircle(getWidth()/2,getHeight()/2,200,paint);
            }
            else if(shape.equals("rect")){
                paint.setColor(Color.GREEN);
                canvas.drawRect(200,400,getWidth()-200,getHeight()-400,paint);
            }
            else if(shape.equals("square")){
                paint.setColor(Color.YELLOW);
                canvas.drawRect(300,500,900,1100,paint);
            }
            else if(shape.equals("text")){
                paint.setColor(Color.LTGRAY);
                paint.setStyle(Paint.Style.FILL);
                canvas.drawText("MANAN", getWidth()/2-150, getHeight()/2, paint);
                paint.setStyle(Paint.Style.STROKE);
            }
        }
    }
}
